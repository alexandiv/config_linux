GRUB_FONT=/boot/grub/fonts/ter-u32n.pf2
GRUB_DISABLE_OS_PROBER=true

GRUB_GFXMODE=auto
GRUB_GFXPAYLOAD_LINUX=keep

menuentry "Arch" {
        insmod part_gpt
        insmod btrfs
        insmod ext2
        set root='hd0,gpt6'
        configfile /boot/grub/grub.cfg
}

menuentry "Windows 11" {
        savedefault
        insmod part_gpt
        insmod fat
        set root=(hd0,gpt1)
        chainloader (${root})/EFI/Microsoft/Boot/bootmgfw.efi
}
